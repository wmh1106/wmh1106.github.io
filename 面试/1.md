## 21 写一个能遍历对象和数组的通用 forEach 函数

```js
function forEach(obj, fn) {
  var key;
  if (obj instanceof Array) {
    obj.forEach(function(item, index) {
      fn(index, item);
    });
  } else {
    for (key in obj) {
      if (obj.hasOwnProperty(key)) {
        fn(key, obj[key]);
      }
    }
  }
}
```

## 求n!，用递归来实现

```js
function factor(n){
    var ret = 1;
    if(n===1){
        ret =1;
    }else{
        ret = n * factor(n-1);
    }
    return ret;
}
console.log(factor(3));
```
